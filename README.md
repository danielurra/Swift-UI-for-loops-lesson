## Grab the code
```swift
import UIKit

var favFruits: [String] = ["banana", "mango", "papaya", "ciruela",]


for i in 1...5 {
    
    print(i)
    
}


for fruitsForLoop in favFruits {
    
    print(fruitsForLoop)
    
}
 
for numberInt in 1...3 {
    
    print(numberInt * 3)
}

for numberDouble in 1...3 {
    
    print(Double(numberDouble) / 3.0)
    
}

```
